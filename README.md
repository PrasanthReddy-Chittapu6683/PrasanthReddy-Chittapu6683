```typescript
// Example: Framework-agnostic Lit Web Component
import { LitElement, html, css } from 'lit';
import { customElement, property } from 'lit/decorators.js';

@customElement('org-button')
export class OrgButton extends LitElement {
  @property({ type: String }) variant = 'primary';
  @property({ type: String }) size = 'medium';
  @property({ type: Boolean }) disabled = false;
  
  static styles = css`
    /* Theme-aware styles using CSS variables */
    :host {
      --btn-primary: var(--brand-primary, #6366f1);
      --btn-radius: var(--brand-radius, 8px);
    }
  `;
  
  render() {
    return html`
      <button class="org-btn org-btn--${this.variant} org-btn--${this.size}">
        <slot></slot>
      </button>
    `;
  }
}

// Usage in ANY framework:
// <org-button variant="primary">Click Me</org-button>
```
